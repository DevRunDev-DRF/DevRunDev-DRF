# ERD
- dbdiagram.io

```
// 사용자 관련 테이블
Table User {
  id              int       [pk, increment]
  email           varchar(254) [unique]
  username        varchar(150)
  password        varchar(128)
  role            enum("Student", "Instructor", "manager") [default: "Student"]
  is_verified     boolean   [default: true]
  is_active       boolean   [default: true]
  is_staff        boolean   [default: false]
  date_joined     datetime
}

Table InstructorApplication {
  id               int      [pk, increment]
  user_id          int      [ref: > User.id]
  qualifications   text
  experience       text
  status           enum("PENDING", "APPROVED", "REJECTED") [default: "PENDING"]
  created_at       datetime
}

// 강의 컨텐츠 테이블
Table Course {
  id              int       [pk, increment]
  instructor_id   int       [ref: > User.id]
  title           varchar(255)
  description     text
  price           int       [default: 0]
  status          enum("review", "approved", "not_approved") [default: "review"]
  avg_rating      float     [default: 0.0]
  created_at      datetime
}

Table Section {
  id              int       [pk, increment]
  course_id       int       [ref: > Course.id]
  title           varchar(255)
  order           int
}

Table Lesson {
  id              int       [pk, increment]
  section_id      int       [ref: > Section.id]
  title           varchar(255)
  video_url       varchar(500) [null]
  order           int
}

// 수강 관련 테이블
Table Enrollment {
  id                  int       [pk, increment]
  student_id          int       [ref: > User.id]
  course_id           int       [ref: > Course.id]
  progress            decimal(5,2) [default: 0.00]
  status              enum("in_progress", "completed", "dropped") [default: "in_progress"]
  enrolled_at         datetime
  
  indexes {
    (student_id, course_id) [unique]
  }
}

Table LessonProgress {
  id                  int       [pk, increment]
  student_id          int       [ref: > User.id]
  lesson_id           int       [ref: > Lesson.id]
  completed           boolean   [default: false]
  completed_at        datetime  [null]
}

Table Certificate {
  id                  int       [pk, increment]
  enrollment_id       int       [ref: > Enrollment.id]
  certificate_id      varchar(50) [unique]
  issued_at           datetime
}

// 장바구니 및 결제
Table CartItem {
  id                  int       [pk, increment]
  user_id             int       [ref: > User.id]
  course_id           int       [ref: > Course.id]
  added_at            datetime
}

Table Payment {
  id                  int       [pk, increment]
  user_id             int       [ref: > User.id]
  merchant_uid        varchar(100) [unique]
  amount              int       [default: 0]
  status              enum("ready", "paid", "cancelled", "failed") [default: "ready"]
  created_at          datetime
}

// 퀴즈 시스템
Table Quiz {
  id                  int       [pk, increment]
  title               varchar(200)
  course_id           int       [ref: > Course.id]
  lesson_id           int       [ref: > Lesson.id, null]
  instructor_id       int       [ref: > User.id]
  created_at          datetime
}

Table QuizQuestion {
  id                  int       [pk, increment]
  quiz_id             int       [ref: > Quiz.id]
  text                text
  order               int       [default: 0]
}

// 평가 및 리뷰
Table Review {
  id                  int       [pk, increment]
  user_id             int       [ref: > User.id]
  course_id           int       [ref: > Course.id]
  rating              int
  comment             text
  created_at          datetime
}

// Q&A 시스템
Table QnaQuestion {
  id                  int       [pk, increment]
  user_id             int       [ref: > User.id]
  course_id           int       [ref: > Course.id]
  title               varchar(200)
  content             text
  is_resolved         boolean   [default: false]
  created_at          datetime
}

Table QnaAnswer {
  id                  int       [pk, increment]
  question_id         int       [ref: > QnaQuestion.id]
  user_id             int       [ref: > User.id]
  content             text
  is_accepted         boolean   [default: false]
  created_at          datetime
}
```
